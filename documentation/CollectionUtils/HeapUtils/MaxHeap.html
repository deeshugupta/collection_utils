def extract_max
  return if is_empty?
  maximum = root
  if @heap.size > 1
    last_value = @heap.last
    last_value[:index] = 0
    @heap = @heap.slice(0..@heap.size-2)
    @heap[0] = last_value
    heapify(root)
  else
    @heap =[  ]
  end

  return maximum[:element]
end

def insert(element)
  value = {element: element, index: size}
  @heap << value
  i = @heap.size - 1
  node, index = parent(i)
  while i != 0 && @heap[i][:element] >= node[:element] do
    exchange(@heap[i], node)
    i = index
    node, index = parent(i)
  end
end
